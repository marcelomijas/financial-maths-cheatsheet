\newcommand{\utpoa}{\mathrm{a}_{n \rceil i} = \dfrac{1 - (1 + i)^{-n}}{i}} 
\newcommand{\utpos}{\mathrm{s}_{n \rceil i} = \mathrm{a}_{n \rceil i} \cdot (1 + i)^n}

\newcommand{\gtpoa}{\mathrm{A}(C;q)_{n \rceil i} = \left \{ 
	\begin{array}{cr}
		C \cdot \dfrac{1 - \left( \dfrac{q}{1 + i} \right)^n}{1 + i - q} & \mathrm{if} \quad q \neq 1 + i \\
		\dfrac{C \cdot n}{1 + i}                                         & \mathrm{if} \quad q = 1 + i
	\end{array}
\right . }
\newcommand{\gtpos}{\mathrm{S}(C;q)_{n \rceil i} = \mathrm{A}(C;q)_{n \rceil i} \cdot (1 + i)^n}

\newcommand{\utpra}{\mathrm{\ddot{a}}_{n \rceil i} = (1 + i) \cdot \mathrm{a}_{n \rceil i}}
\newcommand{\utprs}{\mathrm{\ddot{s}}_{n \rceil i} = (1 + i) \cdot \mathrm{s}_{n \rceil i}}

\newcommand{\gtpra}{\mathrm{\ddot{A}}(C;q)_{n \rceil i} = (1 + i) \cdot \mathrm{A}(C;q)_{n \rceil i}}
\newcommand{\gtprs}{\mathrm{\ddot{S}}(C;q)_{n \rceil i} = (1 + i) \cdot \mathrm{S}(C;q)_{n \rceil i}}

\newcommand{\uppoa}{\mathrm{\ddot{a}}_{\infty \rceil i} = \dfrac{1}{i}}
\newcommand{\gppoa}{\mathrm{A}(C;q)_{\infty \rceil i} = \left \{ 
	\begin{array}{cr}
		\dfrac{C}{1 + i - q} & \mathrm{if} \quad q < 1 + i    \\
		\infty               & \mathrm{if} \quad q \geq 1 + i
	\end{array}
\right . }

\newcommand{\uppra}{\mathrm{\ddot{a}}_{\infty \rceil i} = (1 + i) \cdot \mathrm{a}_{\infty \rceil i}}
\newcommand{\gppra}{\mathrm{\ddot{A}}(C;q)_{\infty \rceil i} = (1 + i) \cdot \mathrm{A}(C;q)_{\infty \rceil i}}

\newcommand{\dpv}{\mathrm{discounted present value} \rightarrow V_0 = C \cdot \mathrm{a}}
\newcommand{\cpv}{\mathrm{capitalized present value} \rightarrow V_n = C \cdot \mathrm{s}}
